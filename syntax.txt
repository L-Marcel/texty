<!-- Programa e subprogramas -->
<program> -> <program> <program_slice> | <program_slice>
<program_slice> -> <subprogram> | <enum> | <struct> | <trait> | <impl>
<subprogram> -> <fn> | <proc>

<fn> -> FUNCTION ID <params_list> COLON <type> <stmts> END_FUNCTION SEMICOLON
<proc> -> PROCEDURE ID <params_list> <stmts> END_PROCEDURE SEMICOLON
        | PROCEDURE ID <params_list> END_PROCEDURE SEMICOLON

<params_self_list> -> LEFT_PAREN SELF SEMICOLON <params> RIGHT_PAREN | LEFT_PAREN SELF RIGHT_PAREN
<params_list> -> LEFT_PAREN <params> RIGHT_PAREN | LEFT_PAREN RIGHT_PAREN
<params> -> <params> SEMICOLON <param> | <param>
<param> -> <id_list> COLON <type>
<id_list> -> <id_list> COMMA ID | ID

<subprogram_call> -> <access> <call_params_list>
<call_params_list> -> LEFT_PAREN <call_params> RIGHT_PAREN | LEFT_PAREN RIGHT_PAREN
<call_params> -> <call_params> COMMA <expr> | <expr>

<!-- Enumerações -->
<enum> -> ENUM <name> <enum_values> END_ENUM SEMICOLON
        | ENUM <name> END_ENUM SEMICOLON
<enum_values> -> <enum_values> COMMA CONST_NAME
               | CONST_NAME
  
<!-- Registros -->
<struct> -> STRUCT <name> <struct_attrs> END_STRUCT SEMICOLON
          | STRUCT <name> END_STRUCT SEMICOLON
<struct_attrs> -> <struct_attrs> <struct_attr> SEMICOLON
                | <struct_attr> SEMICOLON
<struct_attr> -> <id_list> COLON <type>

<!-- Características -->
<trait> -> TRAIT <name> <trait_subprograms> END_TRAIT SEMICOLON
         | TRAIT <name> END_TRAIT SEMICOLON
<trait_subprograms> -> <trait_subprograms> <trait_subprogram>
                     | <trait_subprogram>
<trait_subprogram> -> <trait_fn> | <trait_proc>
<trait_fn> -> FUNCTION ID <params_self_list> COLON <type> END_FUNCTION SEMICOLON
            | FUNCTION ID <params_list> COLON <type> END_FUNCTION SEMICOLON
            | <fn>
<trait_proc> -> PROCEDURE ID <params_self_list> END_PROCEDURE SEMICOLON
              | PROCEDURE ID <params_list> END_PROCEDURE SEMICOLON
              | <proc>

<!-- Implementações -->
<impl> -> IMPL <name> <impl_subprograms> END_IMPL SEMICOLON
        | IMPL <name> END_IMPL SEMICOLON
        | IMPL <name> <impl_subprograms> FOR <name> END_IMPL SEMICOLON
        | IMPL <name> FOR <name> END_IMPL SEMICOLON
<impl_subprograms> -> <impl_subprograms> <impl_subprogram>
                     | <impl_subprogram>
<impl_subprogram> -> <impl_fn> | <impl_proc>
<impl_fn> -> FUNCTION ID <params_self_list> COLON <type> <stmts> END_FUNCTION SEMICOLON
           | <fn>
<impl_proc> -> PROCEDURE ID <params_self_list> <stmts> END_PROCEDURE SEMICOLON
             | PROCEDURE ID <params_self_list> END_PROCEDURE SEMICOLON
             | <proc>

<!-- Acesso -->
<access> -> <access> DOT CONST_NAME
          | <access> DOT ID
          | <access> LEFT_BRACKET <number_expr> RIGHT_BRACKET
          | <access> DOUBLE_COLON ID
          | <subprogram_call>
          | <access_base>
<access_base> -> SELF 
               | <id>  
               | MULT <pointer_result>
               | LEFT_PAREN <expr> RIGHT_PAREN

<!-- Declarações -->
<id> -> NAME | ID
<name> -> NAME | CONST_NAME
<stmts> -> <stmts> <stmt> SEMICOLON | <stmt> SEMICOLON
<stmt> -> BREAK
        | CONTINUE 
        | <attr>
        | <assign>
        | <return>
        | <if>
        | <for>
        | <while>
        | <repeat>
        | <switch>
        | <expr>

<return> -> RETURN <expr>
<attr> -> VAR ID COLON <type> ATTR <expr>
<type> -> TYPE | TYPE LEFT_BRACKET RIGHT_BRACKET | POINTER LT <type> GT
<assign> -> <access> ATTR <expr>
          | <access> AND_ATTR <bool_expr>
          | <access> OR_ATTR <bool_expr>
          | <access> LAZY_AND_ATTR <bool_expr>
          | <access> LAZY_OR_ATTR <bool_expr>
          | <access> XOR_ATTR <bool_expr>
          | <access> CONCAT_ATTR <string_expr>
          | <access> MOD_ATTR <number_expr>
          | <access> PLUS_ATTR <number_expr>
          | <access> MINUS_ATTR <number_expr>
          | <access> MULT_ATTR <number_expr>
          | <access> DIV_ATTR <number_expr>

<!-- Estruturas condicionais -->
<if> -> IF <expr> THEN <stmts> <if_end> 
<if_end> -> ELIF <expr> THEN <stmts> <if_end>
          | ELSE <stmts> END_IF  
          | END_IF
<switch> -> SWITCH <expr> <cases> END_SWITCH
<cases> -> <case_list> <default_case>
         | <case_list>
         | <default_case>
<case_list> -> <case_list> <case>
             | <case>
<case> -> CASE <case_values> COLON <stmts>
<default_case> -> DEFAULT COLON <stmts>

<!-- Estruturas de repetição -->
<for> -> FOR ID IN <range_expr> <stmts> END_FOR
       | FOR ID IN <range_expr> END_FOR
       | FOR ID IN <array_expr> <stmts> END_FOR
       | FOR ID IN <array_expr> END_FOR
       | FOR LEFT_PAREN <attr> SEMICOLON <bool_expr> SEMICOLON <expr> RIGHT_PAREN <stmts> END_FOR
       | FOR LEFT_PAREN <attr> SEMICOLON <bool_expr> SEMICOLON <expr> RIGHT_PAREN END_FOR
<while> -> WHILE <bool_expr> <stmts> END_WHILE
         | WHILE <bool_expr> END_WHILE
<repeat> -> REPEAT <stmts> UNTIL <bool_expr>
          | REPEAT UNTIL <bool_expr>

<!-- Expressões -->
<expr> -> <number_expr> 
        | <bool_expr>
        | <string_expr>
        | <range_expr>
        | <array_expr>
        | <pointer_expr>
<comparable_expr> -> <relational_expr>
                   | <string_expr>
                   | <range_expr>
                   | <array_expr>
                   | <pointer_expr>
                   | <struct_expr>
<in_value> -> <number_expr>
            | <string_expr>
            | <range_expr>
            | <array_expr>
            | <pointer_expr>
            | <bool_result>
<in_owner> -> <range_expr>
            | <array_expr>
            | <string_expr>

<!-- Expressões de subfaixas -->
<range_expr> -> RANGE <number_expr>
              | <number_expr> RANGE
              | <number_expr> RANGE <number_expr>
              | <range_left_limit> <number_expr> RANGE <number_expr> <range_right_limit>
              | <range_left_limit> <number_expr> RANGE <number_expr>
              | <number_expr> RANGE <number_expr> <range_right_limit>
<range_left_limit> -> LEFT_PAREN
                    | LEFT_BRACKET
<range_right_limit> -> RIGHT_PAREN
                     | RIGHT_BRACKET

<!-- Expressões de ponteiros -->
<pointer_expr> -> REF <pointer_result>
<pointer_result> -> <access>

<!-- Expressões numéricas -->
<number_expr> -> <number_expr> BAR <number_bitwise_xor_expr>
               | <number_bitwise_xor_expr>
<number_bitwise_xor_expr> -> <number_bitwise_xor_expr> XOR <number_bitwise_and_expr>
                           | <number_bitwise_and_expr>
<number_bitwise_and_expr> -> <number_bitwise_and_expr> REF <sum_expr>
                           | <sum_expr>
<sum_expr> -> <sum_expr> MINUS <mult_expr> 
            | <sum_expr> PLUS <mult_expr>
            | <mult_expr>
<mult_expr> -> <mult_expr> DIV <sub_expr>
             | <mult_expr> MULT <sub_expr>
             | <mult_expr> MOD <sub_expr>
             | <sub_expr>
<sub_expr> -> MINUS <exp_expr>
            | REV <exp_expr>
            | <exp_expr>
<exp_expr> -> <increment_expr> EXP <exp_expr>
            | <increment_expr>
<increment_expr> -> INCREMENT <number_result>
                  | <number_result> INCREMENT
                  | DECREMENT <number_result>
                  | <number_result> DECREMENT
                  | <number_result>
<number_result> -> NUMBER 
               | <access>

<!-- Expressões booleanas -->
<bool_expr> -> <bool_expr> OR <and_expr>
             | <and_expr>
<and_expr> -> <and_expr> AND <bool_bitwise_or_expr>
            | <bool_bitwise_or_expr>
<bool_bitwise_or_expr> -> <bool_bitwise_or_expr> BAR <bool_bitwise_and_expr>
                        | <bool_bitwise_and_expr>
<bool_bitwise_and_expr> -> <bool_bitwise_and_expr> REF <equals_expr>
                         | <equals_expr>
<equals_expr> -> <equals_expr> EQ <comparable_expr>
               | <equals_expr> NOT_EQ <comparable_expr>
               | <comparable_expr>
<relational_expr> -> <number_expr> LT <number_expr>
                   | <number_expr> GT <number_expr>
                   | <number_expr> LT_EQ <number_expr>
                   | <number_expr> GT_EQ <number_expr>
                   | <in_value> IN <in_owner>
                   | <negative_expr>
<negative_expr> -> NOT <bool_result>
                 | <bool_result>
<bool_result> -> BOOL 
               | <access>

<!-- Expressões de texto -->
<string_expr> -> <string_expr> INCREMENT <string_result>
               | <string_result>
<string_result> -> STRING
                 | CHAR
                 | <access>

<!-- Expressões de arranjos -->
<array_expr> -> <array_expr> INCREMENT <array_result>
              | <array_result>
<array_allocation> -> NEW <type> LEFT_BRACKET <number_expr> RIGHT_BRACKET
                    | NEW <type> LEFT_BRACKET NUMBER RIGHT_BRACKET LEFT_BRACE <array_allocation_values> RIGHT_BRACE
<array_allocation_values> -> <array_allocation_values> COMMA <expr> 
                           | <expr>
<array_result> -> <access>
                | <array_allocation>
                
<!-- Expressões de registros -->
<struct_expr> -> <struct_result> 
<struct_allocation> -> NEW <type> LEFT_BRACE <struct_allocation_values> RIGHT_BRACE
                     | NEW <type> LEFT_BRACE RIGHT_BRACE 
<struct_allocation_values> -> <struct_allocation_values> COMMA ID COLON <expr> 
                            | ID COLON <expr>
<struct_result> -> <access>
                 | <struct_allocation>
